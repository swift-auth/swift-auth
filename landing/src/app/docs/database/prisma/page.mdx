export const auth = new SwiftAuth({

    provider: 'postgres',

});`}

## Generate the schema

<div className="mt-3 max-w-[560px] text-sm leading-relaxed text-muted-foreground">
   Run the CLI to generate the <code>schema.prisma</code> file. Unlike Drizzle, Prisma uses a single
   schema for all providers — the CLI adjusts the timestamp types automatically based on your
   configured provider.
</div>

<div className="mt-5 rounded-xl border border-border bg-card overflow-hidden">
   <CodeBlock lang="bash" code={`npx @swift-auth/cli generate --output ./prisma/schema.prisma`} />
</div>

<div className="mt-4 max-w-[560px] text-sm leading-relaxed text-muted-foreground">
   Then generate the Prisma client and push the schema to your database.
</div>

<div className="mt-5 rounded-xl border border-border bg-card overflow-hidden">
   <CodeBlock lang="bash" code={`npx prisma generate\nnpx prisma db push`} />
</div>
