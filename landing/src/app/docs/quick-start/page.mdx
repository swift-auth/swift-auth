## 3. Create your auth instance

<div className="mt-3 max-w-[560px] text-sm leading-relaxed text-muted-foreground">
   Create a file at <code>src/lib/auth.ts</code>. Import your adapter along with your database
   connection and pass both into <code>Authio</code>. Set <code>provider</code> to match your
   database — <code>postgres</code>, <code>mysql</code>, or <code>sqlite</code>.
</div>

<div className="mt-5 rounded-xl border border-border bg-card overflow-hidden">
   <Tabs defaultValue="drizzle" className="w-full">
      <div className="border-border px-4 pt-3">
         <TabsList className="h-auto gap-1.5 bg-transparent p-0">
            <TabsTrigger value="drizzle" className="rounded-md border border-transparent font-mono text-xs data-[state=active]:border-border data-[state=active]:bg-background">Drizzle</TabsTrigger>
            <TabsTrigger value="prisma" className="rounded-md border border-transparent font-mono text-xs data-[state=active]:border-border data-[state=active]:bg-background">Prisma</TabsTrigger>
         </TabsList>
      </div>
      <TabsContent value="drizzle" className="m-0">
         <div className="border-b border-border px-5 py-2.5">
            <span className="font-mono text-xs text-muted-foreground">src/lib/auth.ts</span>
         </div>
         <CodeBlock
            lang="ts"
            code={`import { Authio } from '@authio/core';
import { drizzleAdapter } from '@authio/drizzle';
import { db } from '../db/index.js';

export const auth = new Authio({
  baseUrl: '<http://localhost:8000>',
  emailAndPassword: {
    enabled: true,
  },
  database: drizzleAdapter({
    db,
    provider: 'postgres', // 'postgres' | 'mysql' | 'sqlite'
  }),
});`}
         />
      </TabsContent>
      <TabsContent value="prisma" className="m-0">
         <div className="border-b border-border px-5 py-2.5">
            <span className="font-mono text-xs text-muted-foreground">src/lib/auth.ts</span>
         </div>
         <CodeBlock
            lang="ts"
            code={`import { Authio } from '@authio/core';
import { prismaAdapter } from '@authio/prisma';
import { PrismaPg } from '@prisma/adapter-pg';
import { PrismaClient } from '../generated/prisma/client.js';

const adapter = new PrismaPg({ connectionString: process.env.DATABASE_URL! });
const prisma = new PrismaClient({ adapter });

export const auth = new Authio({
  baseUrl: '<http://localhost:8000>',
  emailAndPassword: {
    enabled: true,
  },
  database: prismaAdapter({
    db: prisma,
    provider: 'postgres', // 'postgres' | 'mysql' | 'sqlite'
  }),
});`}
         />
      </TabsContent>
   </Tabs>
</div>

## 4. Generate the database schema

<div className="mt-3 max-w-[560px] text-sm leading-relaxed text-muted-foreground">

   this command overwrites whatever is there.

<div className="mt-5 rounded-xl border border-border bg-card overflow-hidden">

<div className="mt-4 max-w-[560px] text-sm leading-relaxed text-muted-foreground">

import { auth } from './lib/auth.js';

const app = express();

app.use(cookieParser());
app.use(express.json());
app.use(toNodeHandler(auth));

app.listen(8000, () => {
console.log('Server running on port 8000');
});`}
